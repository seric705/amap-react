使用#

loadAmap,
loadPlugins,
useAmap,
usePlugins,
useAmapComponent,
} from '@amap/amap-react';

API#

函数签名备注
loadAmap(): Promise<AMap>可用于自行调用AMap功能
loadPlugins(plugins: string, string[]): Promise<AMap>可用于自行加载插件
useAmap(): Nullable<AMap.Map>可用于自定义扩展组件
usePlugins(plugins: string, string[]) : Promise<AMap>可用于自定义扩展组件
useAmapComponent<T>(init, plugins): Nullable<T>可用于自定义扩展组件
useAmapEvents<E = any>(events, handler)可用于自定义扩展组件
